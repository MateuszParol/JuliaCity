function oblicz_energie(D::Matrix{Float64}, trasa::Vector{Int}, bufor::Vector{Float64})::Float64
    n = length(trasa)
    nchunks = length(bufor)
    fill!(bufor, 0.0)
    # BL-04 fix (gap-closure 02-10): kanoniczny pattern - materialized chunks + indexable.
    # `collect(chunks(...))` daje Vector{UnitRange{Int}} (jednorazowy alloc O(nchunks),
    # ≤ Threads.nthreads()), ktory jest indexable zgodnie z @threads :static wymogiem
    # (Julia 1.10 manual). Pre-fix `enumerate(chunks(...))` byl non-canonical i
    # silently broken na Julia minor-version bumpach.
    cs = collect(chunks(1:n; n=nchunks))
    Threads.@threads :static for chunk_idx in eachindex(cs)
        s = 0.0
        @inbounds for k in cs[chunk_idx]
            i_aktualne = trasa[k]
            i_nastepne = trasa[mod1(k + 1, n)]
            s += D[i_aktualne, i_nastepne]
        end
        bufor[chunk_idx] = s
    end
    return sum(bufor)
end

     - WR-07 zazyly w teste ENE-03 ma threshold `< 4096`, wiec ten alloc miesci sie.
