```julia
function delta_energii(stan, i, j)
    n = length(stan.trasa)
    i_next = i + 1                  # i+1 <= j, wiec bez wrap
    j_next = mod1(j + 1, n)         # jedyny edge dla j=n
    t = stan.trasa
    D = stan.D
    return  D[t[i],      t[j]]      + D[t[i_next], t[j_next]] -
            D[t[i],      t[i_next]] - D[t[j],      t[j_next]]
end
```

```julia
using ChunkSplitters
function oblicz_energie(D, trasa, bufor)
    n = length(trasa)
    fill!(bufor, 0.0)
    Threads.@threads :static for (chunk_idx, krawedzie) in enumerate(chunks(1:n; n=Threads.nthreads()))
        s = 0.0
        @inbounds for k in krawedzie
            s += D[trasa[k], trasa[mod1(k+1, n)]]
        end
        bufor[chunk_idx] = s
    end
    return sum(bufor)
end
```

From RESEARCH Example 1 (oblicz_macierz_dystans! pattern):
```julia
function oblicz_macierz_dystans!(stan::StanSymulacji)
    n = length(stan.punkty)
    @assert size(stan.D) == (n, n) "D dimension mismatch"
    @inbounds for j in 1:n
        for i in 1:j-1
            p_i = stan.punkty[i]; p_j = stan.punkty[j]
            dx = p_i[1] - p_j[1]; dy = p_i[2] - p_j[2]
            d = sqrt(dx*dx + dy*dy)
            stan.D[i,j] = d; stan.D[j,i] = d
        end
        stan.D[j,j] = 0.0
    end
    return nothing
end
```

From RESEARCH Example 2 (kalibruj_T0 pattern):
```julia
function kalibruj_T0(stan::StanSymulacji; n_probek::Int=1000, rng=stan.rng)
    n = length(stan.trasa)
    worsening = Float64[]
    sizehint!(worsening, n_probek)
    for _ in 1:n_probek
        i = rand(rng, 1:(n - 1))
        j = rand(rng, (i + 2):n)
        delta = delta_energii(stan, i, j)
        if delta > 0.0
            push!(worsening, delta)
        end
    end
    @assert !isempty(worsening) "no worsening moves sampled"
    sigma = std(worsening)
    return 2.0 * sigma
end
```
