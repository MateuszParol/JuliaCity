```julia
function _live_loop(fig, stan, params, alg, obs_trasa, obs_ene_historia,
                    obs_overlay, liczba_krokow, fps, kroki_na_klatke)
    # Rolling window dla FPS i accept rate
    fps_window   = zeros(Float64, 60)   # ostatnie 60 timestampow
    acc_window   = zeros(Bool, 1000)    # ostatnie 1000 krokow
    fps_idx      = Ref(0)
    acc_idx      = Ref(0)
    t_prev       = Ref(time())

    while isopen(fig) && stan.iteracja < liczba_krokow
        for _ in 1:kroki_na_klatke
            energia_przed = stan.energia
            symuluj_krok!(stan, params, alg)
            acc_window[mod1(acc_idx[] += 1, 1000)] = (stan.energia <= energia_przed)
        end

        # Update Observables (jeden raz na kroki_na_klatke krokow — D-05, VIZ-05)
        obs_trasa[] = _trasa_do_punkty(stan)
        push!(obs_ene_historia[], Point2f(stan.iteracja, stan.energia))
        notify(obs_ene_historia)

        # Overlay strings update
        t_now = time()
        fps_est = 1.0 / max(t_now - t_prev[], 1e-9)
        t_prev[] = t_now
        eta = (liczba_krokow - stan.iteracja) / (fps_est * kroki_na_klatke)
        acc_rate = sum(acc_window) / min(acc_idx[], 1000)
        obs_overlay[] = _zbuduj_overlay_string(stan, alg, fps_est, eta, acc_rate)

        sleep(1/fps)   # yield do renderloop GLMakie — kluczowe dla responsywnosci
    end
end
```

    ```julia
    function _live_loop(fig, stan::StanSymulacji, params::Parametry, alg::Algorytm,
                        obs_trasa::Observable{Vector{Point2f}},
                        obs_historia::Observable{Vector{Point2f}},
                        obs_overlay::Observable{String};
                        liczba_krokow::Int, fps::Int, kroki_na_klatke::Int)
        # Rolling windows dla statystyk wyswietlanych w overlay (D-04).
        # FPS — instantaneous z roznicy time() miedzy klatkami (1 sample sufficient
        # dla numerical sensowny avg gdy renderloop jest stabilny).
        # Accept rate — ostatnie 1000 krokow, accepted = (energia po <= energia przed).
        const ACC_WIN = 1000
        acc_window  = falses(ACC_WIN)
        acc_idx     = Ref(0)
        t_prev      = Ref(time())

        # Stop conditions: window zamkniete LUB SA osiagnal liczba_krokow.
        # `isopen(fig)` — z Makie events; jezeli nie wspierane dla Figure (A2),
        # fallback: events(fig).window_open[]. Test w manual smoke per D-14.
        while isopen(fig) && stan.iteracja < liczba_krokow
            # 1. SA steps (throttling per D-05): kroki_na_klatke krokow per klatka.
            for _ in 1:kroki_na_klatke
                stan.iteracja >= liczba_krokow && break
                energia_przed = stan.energia
                symuluj_krok!(stan, params, alg)
                acc_idx[] += 1
                acc_window[mod1(acc_idx[], ACC_WIN)] = (stan.energia <= energia_przed)
            end

            # 2. Update Observables (jeden notify per Observable per klatka — VIZ-05).
            obs_trasa[] = _trasa_do_punkty(stan)
            # obs_historia: push! na .val + recznym notify dla unikniecia O(n) realloc.
            push!(obs_historia.val, Point2f(Float32(stan.iteracja), Float32(stan.energia)))
            notify(obs_historia)

            # 3. Liczenie statystyk + update overlay (D-04 — wszystkie 7 pol).
            t_now = time()
            dt = max(t_now - t_prev[], 1e-9)
            fps_est = 1.0 / dt
            t_prev[] = t_now

            # ETA: pozostale klatki * sekundy/klatke.
            kroki_pozostale = liczba_krokow - stan.iteracja
            klatki_pozostale = max(0, kroki_pozostale ÷ kroki_na_klatke)
            eta_sec = klatki_pozostale * dt   # uzywamy ostatniej dt — proxy dla fps stabilny

            # Accept rate (rolling 1000 — przed osiagnieciem 1000 krokow uzywamy actual count).
            n_samples = min(acc_idx[], ACC_WIN)
            acc_rate = n_samples > 0 ? count(acc_window[1:n_samples]) / n_samples : NaN

            obs_overlay[] = _zbuduj_overlay_string(stan, alg, fps_est, eta_sec, acc_rate)

            # 4. Yield do GLMakie event loop (RESEARCH Q2 + Pitfall C).
            #    Bez tego okno staje sie "Not Responding".
            sleep(1/fps)
        end

        return nothing
    end
    ```

Jezeli executor wybiera nie-uzywac tej konstrukcji (kontrowersyjne — niektore wersje Julii moga sie skarzyc), uzyc `ACC_WIN = 1000` (bez const, ale wewnatrz funkcji to i tak jest constant per invocation).
